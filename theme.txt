  function playSynthChime(frequency, type) {
    if (soundMuted) return;
    try {
      if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
      if (audioCtx.state === 'suspended') audioCtx.resume();
      var osc = audioCtx.createOscillator();
      var gain = audioCtx.createGain();
      osc.type = type || 'sine';
      osc.frequency.setValueAtTime(frequency || 440, audioCtx.currentTime);
      gain.gain.setValueAtTime(0.15, audioCtx.currentTime);
      gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.8);
      osc.connect(gain);
      gain.connect(audioCtx.destination);
      osc.start();
      osc.stop(audioCtx.currentTime + 0.8);
    } catch (e) {}
  }

document.addEventListener('DOMContentLoaded', function() {
  document.querySelectorAll('.cms-dynamic-widget').forEach(function(widget) {
    var label = widget.getAttribute('data-label');
    var max = widget.getAttribute('data-max') || 4;
    if (!label) return;
    fetch('/feeds/posts/summary/-/' + encodeURIComponent(label) + '?alt=json&max-results=' + max)
      .then(function(res) { return res.json(); })
      .then(function(data) {
        var entries = (data.feed && data.feed.entry) ? data.feed.entry : [];
        if (entries.length === 0) {
          widget.innerHTML = '<p style="text-align:center;color:var(--gray)">No posts found in ' + label + '</p>';
          return;
        }
        var html = '<div class="blog-posts-grid">';
        entries.forEach(function(entry) {
          var title = entry.title.$t;
          var url = (entry.link || []).find(function(l) { return l.rel === 'alternate'; });
          url = url ? url.href : '#';
          var imgRaw = entry.media$thumbnail ? entry.media$thumbnail.url : '';
          var img = imgRaw ? imgRaw.replace('/s72-c', '/s400') : '';
          var snippet = entry.summary ? entry.summary.$t.replace(/<[^>]*>/g, '').substring(0, 100) + '...' : '';
          html += '<div class="blog-post-card fade-in">';
          html += img ? '<img class="post-featured-img" src="' + img + '" alt="' + title.replace(/"/g, '&quot;') + '"/>' : '<div class="post-gradient-placeholder" style="background:linear-gradient(135deg,#ffb6b9,#d5a6bd,#a1e3d8)">&#127925;</div>';
          html += '<div class="post-content"><h2><a href="' + url + '">' + title + '</a></h2>';
          html += '<div class="post-snippet">' + snippet + '</div>';
          html += '<a class="read-more-link" href="' + url + '">Watch/Read <i class="fas fa-arrow-right"></i></a></div></div>';
        });
        html += '</div>';
        widget.innerHTML = html;
        // Re-observe new elements for fade-in animation
        if ('IntersectionObserver' in window) {
          var obs = new IntersectionObserver(function(entries) {
            entries.forEach(function(entry) { if (entry.isIntersecting) { entry.target.classList.add('visible'); obs.unobserve(entry.target); } });
          }, { threshold: 0.1 });
          widget.querySelectorAll('.fade-in').forEach(function(el) { obs.observe(el); });
        } else {
          widget.querySelectorAll('.fade-in').forEach(function(el) { el.classList.add('visible'); });
        }
      })
      .catch(function(err) {
        widget.innerHTML = '<p style="text-align:center;color:var(--gray)">Videos loading failed. <a href="/search">Browse all posts</a></p>';
      });
  });
});
